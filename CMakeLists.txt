cmake_minimum_required(VERSION 3.18)
project(pathTracer LANGUAGES CXX CUDA)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CUDA_STANDARD 17)
set(CMAKE_CUDA_STANDARD_REQUIRED ON)

# 1. Explicitly locate the modern CUDA Toolkit targets
find_package(CUDAToolkit REQUIRED)

# 2. Find graphics dependencies
find_package(PkgConfig REQUIRED)
pkg_check_modules(GLFW REQUIRED glfw3)
find_package(GLEW REQUIRED)

# 3. Define the executable
add_executable(path_tracer main.cpp kernel.cu)

set_target_properties(path_tracer PROPERTIES CUDA_ARCHITECTURES "89")

# 4. Link using the official modern targets discovered by find_package(CUDAToolkit)
target_link_libraries(path_tracer PRIVATE 
    CUDA::cudart                  # This target is safely generated by find_package(CUDAToolkit)
    GLEW::GLEW 
    ${GLFW_LIBRARIES} 
    GL
)
